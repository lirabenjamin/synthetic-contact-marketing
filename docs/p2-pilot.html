# Helper functions
pformat <- function(p, digits = 3) {
  threshold <- 10^(-digits)
  if (p < threshold) paste0("< .", paste(rep("0", digits - 1), collapse = ""), "1")
  else paste0("= ", sprintf(paste0("%.", digits, "f"), p))
}

numformat <- function(x, digits = 2) sprintf(paste0("%.", digits, "f"), x)

Accuracy direction breakdown

Code 
# Classify pre-interaction bias direction and whether accuracy improved
acc_direction <- dat %>%
  mutate(
    bias_pre = green_outgroup_pre - actual_green,
    bias_post = green_outgroup_post - actual_green,
    bias_direction = case_when(
      bias_pre > 0 ~ "Overestimate",
      bias_pre < 0 ~ "Underestimate",
      TRUE ~ "Exact"
    ),
    acc_change = accuracy_post - accuracy_pre,
    acc_outcome = case_when(
      acc_change > 0 ~ "Improved",
      acc_change < 0 ~ "Worsened",
      TRUE ~ "No change"
    )
  )

# Summary table
acc_dir_table <- acc_direction %>%
  group_by(learner_party, bias_direction) %>%
  summarise(
    n = n(),
    n_improved = sum(acc_outcome == "Improved"),
    n_worsened = sum(acc_outcome == "Worsened"),
    n_nochange = sum(acc_outcome == "No change"),
    pct_improved = n_improved / n * 100,
    pct_worsened = n_worsened / n * 100,
    pct_nochange = n_nochange / n * 100,
    mean_acc_change = mean(acc_change, na.rm = TRUE),
    mean_bias_pre = mean(bias_pre),
    mean_bias_post = mean(bias_post),
    .groups = "drop"
  )

acc_dir_tests <- acc_direction %>%
  group_by(learner_party, bias_direction) %>%
  summarise(
    p_value = tryCatch(t.test(acc_change, mu = 0)$p.value, error = function(e) NA_real_),
    .groups = "drop"
  )

acc_dir_table <- acc_dir_table %>%
  left_join(acc_dir_tests, by = c("learner_party", "bias_direction"))

acc_dir_table %>%
  mutate(
    `Significant Improvement?` = case_when(
      is.na(p_value) ~ "NA",
      p_value < .05 & mean_acc_change > 0 ~ "Yes",
      TRUE ~ "No"
    ),
    `p (Improvement)` = map_chr(p_value, ~ if (is.na(.x)) "NA" else pformat(.x)),
    across(c(pct_improved, pct_worsened, pct_nochange, mean_acc_change, mean_bias_pre, mean_bias_post), ~ numformat(.x))
  ) %>%
  select(
    Party = learner_party,
    `Pre-Bias Direction` = bias_direction,
    N = n,
    Improved = n_improved,
    `% Improved` = pct_improved,
    Worsened = n_worsened,
    `% Worsened` = pct_worsened,
    `No Change` = n_nochange,
    `% No Change` = pct_nochange,
    `Mean Accuracy Change` = mean_acc_change,
    `Significant Improvement?`,
    `p (Improvement)`,
    `Mean Bias (Pre)` = mean_bias_pre,
    `Mean Bias (Post)` = mean_bias_post
  ) %>%
  gt() %>%
  tab_header(title = "Accuracy Change by Pre-Interaction Bias Direction") %>%
  tab_footnote(
    "Overestimate = believed outgroup held more extreme attitudes than they actually do. Underestimate = believed outgroup held less extreme attitudes. Mean Accuracy Change > 0 indicates improvement. p-values test subgroup change against 0.",
    locations = cells_title()
  )

# Compute correlations and p-values by party
get_cor_p <- function(data) {
  d <- data %>% select(-learner_party)
  r_mat <- cor(d, use = "pairwise.complete.obs")
  p_mat <- matrix(NA, ncol(d), ncol(d))
  for (i in 1:ncol(d)) {
    for (j in 1:ncol(d)) {
      if (i != j) p_mat[i, j] <- cor.test(d[[i]], d[[j]])$p.value
    }
  }
  list(r = r_mat, p = p_mat)
}

make_scatter <- function(data, x, y, xlab, ylab, jitter_y = FALSE) {
  # Test significance by party to style lines
  dr_data <- data %>% filter(learner_party == "D→R")
  rd_data <- data %>% filter(learner_party == "R→D")
  dr_p <- tryCatch(cor.test(dr_data[[x]], dr_data[[y]])$p.value, error = function(e) 1)
  rd_p <- tryCatch(cor.test(rd_data[[x]], rd_data[[y]])$p.value, error = function(e) 1)

  p <- ggplot(data, aes(x = .data[[x]], y = .data[[y]], color = learner_party))
  if (jitter_y) {
    p <- p + geom_jitter(alpha = 0.5, size = 2, height = 0.05, width = 0)
  } else {
    p <- p + geom_point(alpha = 0.5, size = 2)
  }

  # Add lines per party with styling based on significance
  dr_lw <- if (dr_p < .05) 1.2 else 0.5
  dr_alpha <- if (dr_p < .05) 0.9 else 0.3
  rd_lw <- if (rd_p < .05) 1.2 else 0.5
  rd_alpha <- if (rd_p < .05) 0.9 else 0.3

  # Overall relationship (grey line)
  overall_p <- tryCatch(cor.test(data[[x]], data[[y]])$p.value, error = function(e) 1)
  overall_lw <- if (overall_p < .05) 1.0 else 0.5
  overall_alpha <- if (overall_p < .05) 0.6 else 0.25

  p <- p +
    geom_smooth(data = data, aes(x = .data[[x]], y = .data[[y]]),
                method = "lm", se = FALSE, inherit.aes = FALSE,
                linewidth = overall_lw, alpha = overall_alpha, color = "gray50") +
    geom_smooth(data = dr_data, method = "lm", se = FALSE,
                linewidth = dr_lw, alpha = dr_alpha, color = party_colors["D→R"]) +
    geom_smooth(data = rd_data, method = "lm", se = FALSE,
                linewidth = rd_lw, alpha = rd_alpha, color = party_colors["R→D"]) +
    scale_color_manual(values = party_colors) +
    labs(x = xlab, y = ylab) +
    theme_minimal(base_size = 10) +
    theme(legend.position = "none")

  p
}

make_prepost_panel <- function(long_data, means_data, y_var, y_lab, panel_title,
                                sig_label, int_label, dr_sig, rd_sig) {
  # Nudge direction: D→R labels go left, R→D labels go right
  means_data <- means_data %>%
    mutate(
      nudge_x = ifelse(learner_party == "D→R", -0.15, 0.15),
      hjust = ifelse(learner_party == "D→R", 1, 0)
    )

  y_range <- range(long_data[[y_var]], na.rm = TRUE)
  y_top <- y_range[2] + diff(y_range) * 0.15

  # Midpoint of each party's line for star placement
  dr_mid_y <- means_data %>% filter(learner_party == "D→R") %>% summarise(mid = mean(m)) %>% pull(mid)
  rd_mid_y <- means_data %>% filter(learner_party == "R→D") %>% summarise(mid = mean(m)) %>% pull(mid)

  # Star annotation data (placed above the line midpoint)
  star_offset <- diff(y_range) * 0.08
  star_dat <- tibble(
    x = c(1.5, 1.5),
    y = c(dr_mid_y + star_offset, rd_mid_y + star_offset),
    label = c(dr_sig, rd_sig),
    learner_party = c("D→R", "R→D")
  )

  ggplot(long_data, aes(x = time, y = .data[[y_var]], group = ResponseId, color = learner_party)) +
    geom_line(alpha = 0.12) +
    geom_point(alpha = 0.15, size = 0.8) +

    geom_line(data = means_data,
              aes(x = time, y = m, group = learner_party, color = learner_party),
              linewidth = 1.3, inherit.aes = FALSE) +
    # Mean ± SE
    geom_pointrange(data = means_data,
                    aes(x = time, y = m, ymin = m - se, ymax = m + se,
                        group = learner_party, color = learner_party),
                    size = 0.5, linewidth = 0.7, inherit.aes = FALSE) +
    # Mean value labels
    geom_text(data = means_data,
              aes(x = time, y = m, label = numformat(m), color = learner_party),
              nudge_x = means_data$nudge_x, hjust = means_data$hjust,
              size = 2.8, show.legend = FALSE, inherit.aes = FALSE) +
    # Color-coded by-party significance stars above each line
    geom_text(data = star_dat,
              aes(x = x, y = y, label = label, color = learner_party),
              size = 3.5, fontface = "bold", show.legend = FALSE, inherit.aes = FALSE) +

    annotate("text", x = 1.5, y = y_top, label = paste0("time: ", sig_label),

    labs(y = y_lab, x = NULL, title = panel_title) +

}
